\def\makecover{
   	\phantomsection   % 创建一个虚拟的章节锚点
    \pdfbookmark[-1]{\@mytitle}{mytitle}   % 在PDF书签中添加一个顶级书签（-1表示顶级），书签名称为中文标题
    \begin{titlepage}
      	\vspace*{8pt}
      	\begin{center}
  			\begin{figure}[h]
  				\centering
  			  \includegraphics[width=0.6\textwidth]{figures/uirlogo.png}
  			\end{figure}
  			\vspace*{40pt}
  			\hei\xiaochu{本科生毕业论文}
  			\vspace*{35pt}			

            % 3. 中文标题 - 左对齐，黑体三号
  			\begin{flushleft}
  			    \hei\sanhao{中文题目：\@mytitle}
  			\end{flushleft}
  			% \vspace*{2pt}
  			
  			% 4. 外文标题 - 左对齐，NewRoman三号
  			\begin{flushleft}
  			    \fontfamily{ptm}\selectfont\sanhao{英文题目：\@myetitle}
  			\end{flushleft}
  			\vspace*{25pt}
            
  			\vspace*{3pt}
     
            % 5. 信息表格 - 宋体三号
  			\renewcommand\arraystretch{1.5}  % 增加行距
  			\def\@lwidth{3.5cm}    % 左列宽度
            \def\@rwidth{5cm}   % 右列宽度
  			{\song\sanhao{
                \begin{tabular}{l@{\hspace{1.5cm}}c}  % 第一列左对齐、中间间隔2cm、第二列居中
                    % \makebox[宽度][对齐方式]{内容}
                    % 用\makebox固定宽度，实现不同字数两端对齐
                    \makebox[\@lwidth][s]{届别：} & \underline{\makebox[\@rwidth][c]{\@mygrade}} \\
                    \makebox[\@lwidth][s]{院系：} & \underline{\makebox[\@rwidth][c]{\@myfaculty}} \\
                    \makebox[\@lwidth][s]{专业与方向：} & \underline{\makebox[\@rwidth][c]{\@mysubject}} \\
                    \makebox[\@lwidth][s]{姓名：} & \underline{\makebox[\@rwidth][c]{\@myname}} \\
                    \makebox[\@lwidth][s]{学号：} & \underline{\makebox[\@rwidth][c]{\@mynumber}} \\
                    \makebox[\@lwidth][s]{指导教师：} & \underline{\makebox[\@rwidth][c]{\@mysupervisor}} \\
                \end{tabular}
            }}
            
  			\vspace*{40pt}
  			
  			% 6. 完成时间 - 居中，黑体三号
  			\hei\sanhao{\textbf{完成时间：\@mydate}}
            
		\end{center}
	\end{titlepage}
}
